#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    int A[n][n];
   int i,j;
   for(i=0; i<n; i++){
       for(j=0; j<n; j++){
           A[i][j]=(i+1)*(j+1);
           cout << A[i][j]  << " ";
       }    cout << endl;
   } cout << endl;
    return 0;
}
