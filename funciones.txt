#include <iostream>
using namespace std;
int suma (int num1, int num2){
    return num1 + num2;
}
int resta(int num1, int num2){
    return num1 - num2;
}
int multi(int num1, int num2){
    return num1*num1;
}
int main() {
   int num1, num2, S, R, M;
   cin >> num1 >> num2;
   S=suma(num1, num2);
   R=resta(num1,num2);
   M=multi( num1,num2);
   cout << "La suma es : "<< S << endl;
   cout << "La resta es: "<< R << endl;
   cout << "La multiplicacion es :"<< M << endl;

   
   
   


    return 0;
}
